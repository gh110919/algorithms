### Алгоритм поразрядной сортировки (Radix Sort)

###### Алгоритм сортировки, который сортирует числа, сравнивая их отдельные разряды.

```ts
/* Функция сортировки Радикс */
function radixSort(arr: number[]): number[] {
  /* Определение максимального числа в массиве */
  const maxNum = Math.max(...arr) * 10;

  /* Инициализация делителя */
  let divisor = 10;

  /* Повторение цикла до тех пор, пока делитель меньше максимального числа */
  while (divisor < maxNum) {
    /* Создание пустых корзин */
    const buckets = [...Array(10)].map(() => []);

    /* Распределение чисел по корзинам в зависимости от текущего разряда */
    for (let num of arr) {
      buckets[Math.floor((num % divisor) / (divisor / 10))].push(num);
    }

    /* Объединение всех корзин в один массив */
    arr = [].concat(...buckets);

    /* Увеличение делителя в 10 раз для перехода к следующему разряду */
    divisor *= 10;
  }

  /* Возврат отсортированного массива */
  return arr;
}

/* Тестовые кейсы */
console.log(
  radixSort([170, 45, 75, 90, 802, 24, 2, 66])
); /* [2, 24, 45, 66, 75, 90, 170, 802] */
console.log(radixSort([3, 6, 1, 8, 10, 1, 4])); /* [1, 1, 3, 4, 6, 8, 10] */
console.log(
  radixSort([100, 99, 98, 97, 96, 95])
); /* [95, 96, 97, 98, 99, 100] */
```
