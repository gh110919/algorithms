### Поиск в ширину (Breadth-First Search, BFS)

###### Алгоритм поиска в графе, который сначала исследует все вершины на текущем уровне перед переходом к вершинам на следующем уровне.

```ts
function bfs(graph: Map<number, number[]>, start: number): number[] {
  /* Множество для хранения посещенных узлов */
  const visited = new Set<number>();

  /* Очередь для обработки узлов */
  const queue = [start];

  /* Массив для результата обхода */
  const result = [];

  /* Пока очередь не пуста */
  while (queue.length > 0) {
    /* Извлекаем первый элемент из очереди */
    const node = queue.shift();

    /* Если узел ещё не был посещен */
    if (!visited.has(node)) {
      /* Добавляем узел в множество посещенных */
      visited.add(node);

      /* Добавляем узел в результат */
      result.push(node);

      /* Добавляем всех соседей узла в очередь */
      queue.push(...graph.get(node));
    }
  }

  /* Возвращаем массив посещенных узлов в порядке их обхода */
  return result;
}

/* Пример графа в виде списка смежности */
const graph = new Map<number, number[]>([
  [1, [2, 3]],
  [2, [4]],
  [3, [4, 5]],
  [4, [6]],
  [5, []],
  [6, []],
]);

/* Тестовые кейсы */
/* Тестовый запуск BFS на графе с началом в узле 1 */
console.log(bfs(graph, 1)); /* Ожидаемый результат: [1, 2, 3, 4, 6, 5] */

/* Тестовый запуск BFS на графе с началом в узле 2 */
console.log(bfs(graph, 2)); /* Ожидаемый результат: [2, 4, 6] */

/* Тестовый запуск BFS на графе с началом в узле 3 */
console.log(bfs(graph, 3)); /* Ожидаемый результат: [3, 4, 5, 6] */

/* Тестовый запуск BFS на графе с началом в узле 5 (изолированный узел) */
console.log(bfs(graph, 5)); /* Ожидаемый результат: [5] */

/* Тестовый запуск BFS на пустом графе */
const emptyGraph = new Map<number, number[]>();
console.log(bfs(emptyGraph, 1)); /* Ожидаемый результат: [] */
```
